@dataclass
class CircuitBreakerConfig:
    """Configuration for circuit breaker."""

    reset_timeout: float = 60.0
    half_open_timeout: float = 5.0

class CircuitBreaker:
    """Circuit breaker implementation."""
    
    def __init__(
        self,
        config: CircuitBreakerConfig
    ):
        self._config = config
        self._failures = 0
        self._last_failure = None
        self._state = "closed"
        
    async def execute(
        self,
        func: Callable[..., T],
        *args,
        **kwargs
    ) -> T:
        """Execute function with circuit breaker pattern."""
        if self._state == "open":
            if self._should_reset():
                self._state = "half-open"
            else:
                raise CircuitOpenError()
                
        try:
            result = await func(*args, **kwargs)
            
            if self._state == "half-open":
                self._state = "closed"
                self._failures = 0
                
            return result
            
        except Exception as e:
            self._failures += 1
            self._last_failure = datetime.now()
            
            if self._failures >= self._config.failure_threshold:
                self._state = "open"
                
            raise
            
    def _should_reset(self) -> bool:
        """Check if circuit should be reset."""
        if not self._last_failure:
            return False
            
        elapsed = (datetime.now() - self._last_failure).total_seconds()
        return elapsed >= self._config.reset_timeout

                last_error = e
                
                self._metrics.counter(

                    success="false",
                    attempt=str(attempt)
                )
                
                if attempt < self._config.max_attempts:

                    self._logger.warning(

                            "attempt": attempt,
                            "delay": delay
                        }
                    )
                    await asyncio.sleep(delay)
                    
        raise MaxRetriesExceededError(
            f"Max retries ({self._config.max_attempts}) exceeded"
        ) from last_error
        
    def _calculate_delay(
        self,
        attempt: int
    ) -> float:
        """Calculate delay for next retry."""
        delay = min(
            self._config.initial_delay * (
                self._config.exponential_base ** attempt
            ),
            self._config.max_delay
        )
        
        if self._config.jitter:
            delay *= random.uniform(0.5, 1.5)
            
        return delay

class FallbackStrategy:
    """Fallback strategy implementation."""
    
    def __init__(
        self,
        fallbacks: List[Callable[..., T]]
    ):
        self._fallbacks = fallbacks
        self._metrics = MetricsManager.get_instance()
        self._logger = logging.getLogger(__name__)
        
    async def execute(
        self,
        primary: Callable[..., T],
        *args,
        **kwargs
    ) -> T:
        """Execute with fallback strategy."""
        try:
            return await primary(*args, **kwargs)
        except Exception as primary_error:
            self._logger.warning(
                "Primary execution failed, trying fallbacks",
                extra={"error": str(primary_error)}
            )
            
            for i, fallback in enumerate(self._fallbacks):
                try:
                    result = await fallback(*args, **kwargs)
                    
                    self._metrics.counter(
                        "fallback_operations",
                        1,
                        index=str(i),
                        success="true"
                    )
                    
                    return result
                    
                except Exception as e:
                    self._metrics.counter(
                        "fallback_operations",
                        1,
                        index=str(i),
                        success="false"
                    )
                    
                    self._logger.warning(
                        f"Fallback {i} failed",
                        extra={"error": str(e)}
                    )
                    
            raise AllFallbacksFailedError(
                "All fallback strategies failed"
            ) from primary_error
